class Solution {
public:
    int atmostK(string s, int k) {
        int n = s.size();

        unordered_map<char, int> freqMap;

        int left = 0;
        int right = 0;

        int count = 0;

        while(right < n) {
            freqMap[s[right]]++;

            while(freqMap.size() > k) {
                freqMap[s[left]]--;

                if(freqMap[s[left]] == 0)
                    freqMap.erase(s[left]);
                
                left++;
            }

            count += right - left + 1;
            right++;
        }
        return count;
    }

    int numberOfSubstrings(string s) {
        return atmostK(s, 3) - atmostK(s, 2);
    }
};
